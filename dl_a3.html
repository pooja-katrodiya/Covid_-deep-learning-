# reading data and storing them as numpy arrays in a list

images = []
lung_masks = []
condition = []

for f in folders:
    images_dir_path = data_dir + "/" + f + "/" + "images/"
    masks_dir_path = data_dir + "/" + f + "/" + "lung masks/"
    
    images_list = os.listdir(images_dir_path)
    #masks_list = os.listdir(masks_dir_path)

    for i in images_list:
        img = load_img(images_dir_path + i)
        img_arr = img_to_array(img).astype(int)

        mask = load_img(masks_dir_path + i)
        mask_arr = img_to_array(mask).astype(int)
        
        images.append(img_arr)
        lung_masks.append(mask_arr)

        condition.append(f)

In [118]:
model = Sequential()

model.add(InputLayer(input_shape=X[0].shape))

model.add(Conv2D(32,
                 kernel_size=(2,2),
                 strides=(1,1),
                 padding='same',
                 activation='relu'
                 ))

model.add(Conv2D(64,
                 kernel_size=(3,3),
                 strides=(2,2),
                 padding='same',
                 activation='relu'
                 ))
model.add(MaxPool2D(pool_size=(3,3), padding='same'))
model.add(BatchNormalization())
model.add(Dropout(0.2))

model.add(Conv2D(128,
                 kernel_size=(3,3),
                 strides=(3,3),
                 padding='same',
                 activation='relu'
                 ))
model.add(MaxPool2D(pool_size=(3,3), padding='same'))
model.add(BatchNormalization())
model.add(Dropout(0.2))

model.add(Flatten())
model.add(Dense(128, activation='relu', kernel_initializer=initializer1))
model.add(Dense(64, activation='relu', kernel_initializer=initializer2))

model.add(Dense(outputs, activation='softmax'))

              optimizer='adam',
              metrics=['accuracy'])

AUC-ROC Curve¶

In [156]:
fig, ax = plt.subplots(1,3)

fig.set_figwidth(16)

for i in range(len(cols)):
    fpr, tpr, thresholds = roc_curve(Y_test[:, i], Y_pred[:,i])
    auc_v = auc(tpr, fpr)

    ax[i].plot(fpr, tpr, label='Area = {:.3f}'.format(auc_v))
    ax[i].set_xlabel('False positive rate')
    ax[i].set_ylabel('True positive rate')
    ax[i].set_title(cols[i])
    ax[i].legend(loc='best')

plt.show()
